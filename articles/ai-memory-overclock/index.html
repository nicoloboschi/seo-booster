 1import time
 2import collections
 3
 4class OptimizedMemory:
 5 def __init__(self, cache_size=10, history_depth=100):
 6 self.memory = collections.deque(maxlen=history_depth) # Stores all historical data, with max length
 7 self.cache = collections.OrderedDict() # Stores frequently accessed recent items {key: data}
 8 self.cache_size = cache_size
 9
10 def add_memory(self, key, data):
11 """Adds a new memory, prioritizing recent items for cache."""
12 current_time = time.time()
13 self.memory.append((current_time, key, data)) # deque handles maxlen automatically
14
15 # Update cache with the most recent item
16 if key in self.cache:
17 # If key exists, move it to the end to mark it as recently used
18 self.cache.move_to_end(key)
19 self.cache[key] = data
20 else:
21 # If it's a new key and cache is full, remove the least recently used item
22 if len(self.cache) >= self.cache_size:
23 self.cache.popitem(last=False) # last=False removes the first item (LRU)
24 self.cache[key] = data
25 self.cache.move_to_end(key) # Ensure it's at the end
26
27 def recall(self, key):
28 """Recalls data, checking cache first for faster access."""
29 if key in self.cache:
30 # Simulate faster cache retrieval - this is the "overclocked" path
31 print(f"Cache hit for key: {key} (Fast Recall)")
32 self.cache.move_to_end(key) # Mark as recently used
33 return self.cache[key]
34 else:
35 # Simulate slower retrieval from main memory
36 print(f"Cache miss for key: {key}. Searching main memory...")
37 # Search recent items first within the deque
38 for timestamp, mem_key, mem_data in reversed(self.memory):
39 if mem_key == key:
40 # Add to cache if space is available or if it's a new key
41 if len(self.cache) < self.cache_size:
42 self.cache[key] = mem_data
43 self.cache.move_to_end(key) # Ensure it's at the end
44 return mem_data
45 return None # Not found
46
47## Example Usage
48memory_system = OptimizedMemory(cache_size=5, history_depth=50)
49
50## Adding memories
51print("Adding memories...")
52for i in range(15):
53 memory_system.add_memory(f"item_{i}", f"data_{i}")
54
55print("\nRecalling memories...")
56## First recall of item_5 (likely a cache miss, then added to cache)
57print(f"Recall item_5: {memory_system.recall('item_5')}")
58## Second recall of item_5 (should be a cache hit)
59print(f"Recall item_5 again: {memory_system.recall('item_5')}")
60## Recalling an older item that might have been pushed out of cache

```python
class SimpleMemory:
 def __init__(self):
 # This dictionary acts as a simple memory store.
 # In a real Google long term memory AI system, this would be a sophisticated vector database.
 self.memory_store = {}
 self.next_id = 0

 def add_memory(self, content: str, metadata: dict = None):
 """Adds a piece of information to the memory."""
 entry_id = self.next_id
 self.memory_store[entry_id] = {"content": content, "metadata": metadata or {}}
 self.next_id += 1
 print(f"Memory added with ID {entry_id}: '{content}'")
 return entry_id

 def retrieve_memory(self, query: str, limit: int = 3):
 """
 Simulates retrieving memory based on a query.
 A real system would use embeddings and vector search.
 This simple version just looks for keywords.
 """
 print(f"\nRetrieving memories related to: '{query}'")
 results = []
 # Simple keyword matching for demonstration
 for entry_id, data in self.memory_store.items():
 if query.lower() in data["content"].lower() or query.lower() in data["metadata"].get("tags", "").lower():
 results.append((entry_id, data))
 if len(results) >= limit:
 break

 if not results:
 print("No relevant memories found.")
 return []

 print("Found memories:")
 for entry_id, data in results:
 print(f" - ID: {entry_id}, Content: '{data['content']}', Metadata: {data['metadata']}")
 return results
```

## Query for relevant memory

query_embedding = model.encode([query])

## Calculate cosine similarity between query and memory embeddings
similarities = cosine_similarity(query_embedding, memory_embeddings)[0]

## Find the most similar memory entry
most_similar_index = similarities.argmax()
most_similar_memory = memory_entries[most_similar_index]
confidence_score = similarities[most_similar_index]

print(f"Query: {query}")
print(f"Most relevant memory: '{most_similar_memory}' (Confidence: {confidence_score:.2f})")
